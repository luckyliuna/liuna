题目1：找出单词‘success’里每个字母的出现次数，并写出出现次数最多的单词和次数

function suc() {
    var src = 'success';
    var json = {};
    for (var i = 0; i < src.length; i++) {

        if (!json[src.charAt(i)]) {
            json[src.charAt(i)] = 1
        } else {
            json[src.charAt(i)]++
        }
        console.log(json)
    }
    var mymax = 0;
    var myindex = '';
    for (var i in json) {
        if (json[i] > mymax) {
            mymax = json[i];
            myindex = i;
        }
    }
    document.getElementById("p1").innerHTML = "结果是：" + myindex + "：出现了" + mymax + "次"
}
